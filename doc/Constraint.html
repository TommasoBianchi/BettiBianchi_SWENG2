def check_path(path)
        lv = subject.get_path_value(path)
        rv = value.value
        if lv.is_a? Integer
                rv = rv.to_i
        elsif lv.is_a? DateTime
                rv = rv.to_datetime
        end
        return operator.invoke lv, rv
end
